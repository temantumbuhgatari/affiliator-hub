// ================================================================ // AFFILIATOR HUB — Google Apps Script (Lengkap) // Fitur: Terima submission, anti-duplikat, notifikasi email, // update status, baca data untuk dashboard & tracking // ================================================================ // ── KONFIGURASI ────────────────────────────────────────────────── const CONFIG = { ADMIN_EMAIL: 'emailkamu@gmail.com', SHEET_NAME: 'Laporan Konten', COLUMNS: { TIMESTAMP: 1, SHOPEE: 2, WHATSAPP: 3, PLATFORM: 4, CONTENT_TYPE: 5, LINK: 6, NOTES: 7, STATUS: 8, }, SEND_EMAIL: true, DEFAULT_STATUS: 'Sedang Dicek', }; // ───────────────────────────────────────────────────────────────── // ── HANDLE POST ─────────────────────────────────────────────────── function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === 'updateStatus') { return handleUpdateStatus(data); } // getByUser via POST (dari cek-status.html) — tidak dipakai lagi // tapi tetap disimpan sebagai fallback if (data.action === 'getByUser') { return handleGetByUserPost(data); } return handleNewSubmission(data); } catch (err) { return jsonResponse({ status: 'error', message: err.message }); } } // ── SUBMISSION BARU ─────────────────────────────────────────────── function handleNewSubmission(data) { const sheet = getSheet(); const linkCol = CONFIG.COLUMNS.LINK; const lastRow = sheet.getLastRow(); if (lastRow >= 2) { const existingLinks = sheet .getRange(2, linkCol, lastRow - 1, 1) .getValues() .flat() .map(l => l.toString().trim().toLowerCase()); const incomingLink = (data.link || '').trim().toLowerCase(); if (incomingLink && existingLinks.includes(incomingLink)) { return jsonResponse({ status: 'duplicate', message: 'Link ini sudah pernah dikirimkan sebelumnya.' }); } } sheet.appendRow([ data.timestamp || new Date().toLocaleString('id-ID', { timeZone: 'Asia/Jakarta' }), data.shopee || '', data.whatsapp || '', data.platform || '', data.content_type || '', data.link || '', data.notes || '', CONFIG.DEFAULT_STATUS, ]); if (CONFIG.SEND_EMAIL) { sendNotificationEmail(data); } return jsonResponse({ status: 'ok', message: 'Submission berhasil disimpan.' }); } // ── UPDATE STATUS ───────────────────────────────────────────────── function handleUpdateStatus(data) { const sheet = getSheet(); const rowNum = parseInt(data.row); const newStatus = data.status; if (!rowNum || rowNum < 2) { return jsonResponse({ status: 'error', message: 'Row tidak valid.' }); } sheet.getRange(rowNum, CONFIG.COLUMNS.STATUS).setValue(newStatus); return jsonResponse({ status: 'ok', message: 'Status diperbarui.' }); } // ── HANDLE GET (dengan JSONP support untuk cek-status) ──────────── function doGet(e) { try { const action = e.parameter.action || 'getAll'; const callback = e.parameter.callback || ''; // JSONP callback let result; if (action === 'getAll') { result = handleGetAll(); } else if (action === 'getByUser') { result = handleGetByUser(e.parameter); } else { result = jsonResponse({ status: 'error', message: 'Action tidak dikenal.' }); } // Kalau ada callback = JSONP mode (untuk bypass CORS di browser) if (callback) { const json = JSON.parse(result.getContent()); return ContentService .createTextOutput(callback + '(' + JSON.stringify(json) + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return result; } catch (err) { return jsonResponse({ status: 'error', message: err.message }); } } // ── BACA SEMUA DATA (untuk dashboard admin) ─────────────────────── function handleGetAll() { const sheet = getSheet(); const lastRow = sheet.getLastRow(); if (lastRow < 2) return jsonResponse({ status: 'ok', data: [] }); const values = sheet.getRange(2, 1, lastRow - 1, 8).getValues(); const C = CONFIG.COLUMNS; const data = values.map((row, i) => ({ row: i + 2, timestamp: row[C.TIMESTAMP - 1] || '', shopee: row[C.SHOPEE - 1] || '', whatsapp: row[C.WHATSAPP - 1] || '', platform: row[C.PLATFORM - 1] || '', content_type: row[C.CONTENT_TYPE - 1] || '', link: row[C.LINK - 1] || '', notes: row[C.NOTES - 1] || '', status: row[C.STATUS - 1] || CONFIG.DEFAULT_STATUS, })).reverse(); return jsonResponse({ status: 'ok', data }); } // ── BACA DATA PER USER (GET — untuk JSONP dari cek-status) ──────── function handleGetByUser(params) { const sheet = getSheet(); const lastRow = sheet.getLastRow(); if (lastRow < 2) return jsonResponse({ status: 'ok', data: [] }); const values = sheet.getRange(2, 1, lastRow - 1, 8).getValues(); const C = CONFIG.COLUMNS; const shopeeQuery = (params.shopee || '').toLowerCase().trim(); const waQuery = (params.wa || '').toLowerCase().trim(); const data = values .filter(row => { const shopee = (row[C.SHOPEE - 1] || '').toLowerCase().trim(); const wa = (row[C.WHATSAPP - 1] || '').toLowerCase().trim(); if (shopeeQuery) return shopee === shopeeQuery; if (waQuery) return wa === waQuery; return false; }) .map((row, i) => ({ row: i + 2, timestamp: row[C.TIMESTAMP - 1] || '', shopee: row[C.SHOPEE - 1] || '', whatsapp: row[C.WHATSAPP - 1] || '', platform: row[C.PLATFORM - 1] || '', content_type: row[C.CONTENT_TYPE - 1] || '', link: row[C.LINK - 1] || '', notes: row[C.NOTES - 1] || '', status: row[C.STATUS - 1] || CONFIG.DEFAULT_STATUS, })) .reverse(); return jsonResponse({ status: 'ok', data }); } // ── BACA DATA PER USER via POST (fallback) ──────────────────────── function handleGetByUserPost(data) { const params = { shopee: data.shopee || '', wa: data.wa || '', }; return handleGetByUser(params); } // ── KIRIM EMAIL NOTIFIKASI ──────────────────────────────────────── function sendNotificationEmail(data) { try { const subject = `[Affiliator Hub] Submission baru dari @${data.shopee}`; const body = ` Ada submission konten baru masuk! ━━━━━━━━━━━━━━━━━━━━━━━ DATA AFFILIATOR ━━━━━━━━━━━━━━━━━━━━━━━ Username Shopee : @${data.shopee} WhatsApp : ${data.whatsapp} Waktu : ${data.timestamp} ━━━━━━━━━━━━━━━━━━━━━━━ DETAIL KONTEN ━━━━━━━━━━━━━━━━━━━━━━━ Platform : ${data.platform} Jenis Konten : ${data.content_type} Link Konten : ${data.link} Catatan : ${data.notes || '-'} ━━━━━━━━━━━━━━━━━━━━━━━ Buka Dashboard Admin untuk mengubah status konten ini. `.trim(); GmailApp.sendEmail(CONFIG.ADMIN_EMAIL, subject, body); } catch (err) { console.error('Gagal kirim email:', err.message); } } // ── HELPER ─────────────────────────────────────────────────────── function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(CONFIG.SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(CONFIG.SHEET_NAME); sheet.appendRow([ 'Timestamp', 'Username Shopee', 'WhatsApp', 'Platform', 'Jenis Konten', 'Link Konten', 'Catatan', 'Status' ]); const headerRange = sheet.getRange(1, 1, 1, 8); headerRange.setFontWeight('bold'); headerRange.setBackground('#f97316'); headerRange.setFontColor('#ffffff'); sheet.setFrozenRows(1); } return sheet; } function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // Jalankan sekali untuk inisialisasi sheet function setupSheet() { getSheet(); SpreadsheetApp.getUi().alert('Sheet berhasil dikonfigurasi!'); }
